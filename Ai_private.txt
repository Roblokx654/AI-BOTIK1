const isOwner = name => config.owners.includes(name);
const isMod = name => moderators.has(name) || isOwner(name);

// ======== —Å—Ç–∞—Ä—Ç –∫–ª–∏–µ–Ω—Ç–∞ =========
function startClient() {
    client = new teeworlds.Client(config.host, config.port, config.botNick);


    client.on("connected", () => {
        connected = true;
        console.log(`‚úÖ ${config.botNick} –ø–æ–¥–∫–ª—é—á–µ–Ω –∫ ${config.host}:${config.port}`);
        client.game.Say(`${config.botNick} –æ–Ω–ª–∞–π–Ω!`);
    });

    client.on("disconnect", (reason) => {
        connected = false;
        console.log(`‚ùå –û—Ç–∫–ª—é—á–µ–Ω–æ: ${reason}`);
    });

    client.on("message", async (msg) => {
        try {
            const text = String(msg.message ?? "").trim();
            const author = msg.author?.ClientInfo?.name ?? "Server";
            const ts = new Date().toISOString();

            // —Å–æ—Ö—Ä–∞–Ω—è–µ–º –≤ –ø–∞–º—è—Ç—å
            chatMemory.push({ time: ts, author, text });
            if (chatMemory.length > memoryLimit) chatMemory.shift();

            if (author === config.botNick) return;

            // ---- –∫–æ–º–∞–Ω–¥—ã ----
            if (text.startsWith(prefix)) {
                const parts = text.slice(prefix.length).split(/\s+/);
                const cmd = (parts.shift() || "").toLowerCase();
                const args = parts;

                if (!cmd.startsWith("ai_")) return;
                const owner = isOwner(author);
                const mod = isMod(author);
                if (!mod && !owner) return;

                switch (cmd) {
                    case "ai_on":
                        aiEnabled = true;
                        client.game.Say("–ë–æ—Ç –≤–∫–ª—é—á–µ–Ω!");
                        break;
                    case "ai_off":
                        aiEnabled = false;
                        client.game.Say("–ë–æ—Ç –≤—ã–∫–ª—é—á–µ–Ω.");
                        break;
                    case "ai_memory":
                        if (!args[0]) return client.game.Say(`–ò—Å–ø–æ–ª—å–∑—É–π: ${prefix}ai_memory 100`);
                        const n = parseInt(args[0]);
                        if (!Number.isNaN(n) && n > 0 && n <= config.maxMemoryLimit) {
                            memoryLimit = n;
                            while (chatMemory.length > memoryLimit) chatMemory.shift();
                            client.game.Say(`–ü–∞–º—è—Ç—å: ${memoryLimit} —Å–æ–æ–±—â–µ–Ω–∏–π.`);
                        }
                        break;
                    case "ai_promt":
                    case "ai_prompt":
                        if (!owner) return;
                        const newP = args.join(" ").trim();
                        if (newP.length > 5) {
                            currentPrompt = newP;
                            client.game.Say("üß† –ü—Ä–æ–º—Ç –æ–±–Ω–æ–≤–ª—ë–Ω!");
                        }
                        break;
                    case "ai_exit":
                        if (!owner) return;
                        client.game.Say("‚õî –û—Ç–∫–ª—é—á–∞—é—Å—å...");
                        awaitDisconnectClient();
                        break;
                    case "ai_restart":
                        if (!owner) return;
                        client.game.Say("üîÅ –ü–µ—Ä–µ–∑–∞–ø—É—Å–∫...");
                        chatMemory = [];
                        playerMemory = {};
                        awaitRestartClient();
                        break;
                    default:
                        break;
                }
                return;
            }

            // ---- –æ–±—ã—á–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ ----
            if (!aiEnabled || bannedPlayers.has(author)) return;
            if (!text.toLowerCase().includes(config.botNick.toLowerCase())) return;

            const promptText = text.replace(new RegExp(config.botNick, "ig"), "").trim() || "–ü—Ä–∏–≤–µ—Ç!";
            const contextCount = Math.min(30, chatMemory.length);
            const recent = chatMemory
                .slice(-contextCount)
                .map(m => `[${m.author}]: ${m.text}`)
                .join("\n");

            const messages = [
                { role: "system", content: currentPrompt },
                { role: "user", content: `–ö–æ–Ω—Ç–µ–∫—Å—Ç:\n${recent}\n\n${author}: ${promptText}` }
            ];

            const completion = await groq.chat.completions.create({
                model: config.model,
                messages,
                temperature: config.temperature,
                max_tokens: config.maxTokens
            });

            let reply = completion.choices?.[0]?.message?.content?.trim() || "...";
            reply = reply.replace(/^(\s*(?:–Ω–∏–∫|player)\s*:\s*)/i, "");
            const final = `${author}: ${reply}`.slice(0, config.maxReplyLength);

            setTimeout(() => {
                try {
                    client.game.Say(final);
                } catch (e) {
                    console.error("–û—à–∏–±–∫–∞ –æ—Ç–ø—Ä–∞–≤–∫–∏:", e);
                }
            }, config.replyDelay);

        } catch (err) {
            console.error("–û—à–∏–±–∫–∞ –≤ –æ–±—Ä–∞–±–æ—Ç—á–∏–∫–µ —Å–æ–æ–±—â–µ–Ω–∏—è:", err);
        }
    });
    client.connect();
}

// ======== —É—Ç–∏–ª–∏—Ç—ã ========
function awaitDisconnectClient() {
    return new Promise((resolve) => {
        if (!client) return resolve();
        try {
            client.Disconnect().then(() => {
                client = null;
                connected = false;
                resolve();
            }).catch(() => {
                client = null;
                connected = false;
                resolve();
            });
        } catch {
            client = null;
            connected = false;
            resolve();
        }
    });
}

async function awaitRestartClient() {
    await awaitDisconnectClient();
    setTimeout(() => startClient(), 1000);
}

// ======== —Å—Ç–∞—Ä—Ç ========
startClient();

process.on("SIGINT", async () => {
    console.log("‚èπ –ó–∞–≤–µ—Ä—à–µ–Ω–∏–µ...");
    await awaitDisconnectClient();
    process.exit(0);
});
